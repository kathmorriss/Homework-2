Sub StockByYear()    Dim ws As Worksheet    Dim ticker As String    Dim volume As Double    Dim row As Integer    For Each ws In Worksheets    ws.Activate    volume = 0    row = 2        ws.Range("I1").Value = "Ticker"    ws.Range("J1").Value = "Total Stock Volume"        LastRow = ws.Cells(Rows.Count, 1).End(xlUp).row            For i = 2 To LastRow            If Cells(i + 1, 1).Value <> Cells(i, 1).Value Then                ticker = Cells(i, 1).Value                                volume = volume + Cells(i, 7).Value                ws.Range("I" & row).Value = ticker                ws.Range("J" & row).Value = volume                row = row + 1                                volume = 0            Else                volume = volume + Cells(i, 7).Value            End If                      Next i    Next wsEnd Sub
